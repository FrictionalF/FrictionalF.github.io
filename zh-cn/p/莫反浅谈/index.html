void Prime(int x){
    mu[1]=1;
    for(int i=2;i<=x;++i){
        if(!st[i]){
            primes[++k]=i;
            mu[i]=-1;
        }
        for(int j=1;primes[j]*i<=x;++j){

    }
}

int solve(int n,int m){//Êï∞ËÆ∫ÂàÜÂùóÊ±ÇËß£

        ans+=(n/l)*(m/l)*(sum[r]-sum[l-1]);
        l=r+1;
    }
    return ans;
}
